character_event = {
	id = unoccupied.2
	desc = "EVTDESCunoccupied.2"
	
	trigger = {
		culture_group = unsettled
		is_ruler = no	
	}

	option = {
		name = "EVTOPTAunoccupied.2" #Yes
		location = {
			ROOT = {
				if = {
					limit = { is_female = no }
					liege = {
						create_character = {
							random_traits = yes
							dynasty = none
							culture = PREVPREV
							religion = PREVPREV
							female = no
							age = 25
						}
					}
				}
				if = {
					limit = { is_female = yes }
					liege = {
						create_character = {
							random_traits = yes
							dynasty = none
							culture = PREVPREV
							religion = PREVPREV
							female = yes
							age = 19
						}
					}
				}
			}
		}	
		clear_wealth = ROOT
		wealth = -500
		death = yes
	}
}

province_event = {
	id = unoccupied.3
	desc = "EVTDESCunoccupied.3"
	picture = "GFX_evt_castle_construction"
	
	is_triggered_only = yes
	
	immediate = {
		capital_holding = {
			add_building = ca_colony_1
			remove_holding_modifier = recently_conquered
			remove_holding_modifier = new_administration
		}
		if = {
			limit = {
				has_province_flag = da_road_ruin
			}
			clr_province_flag = da_road_ruin
			add_province_modifier = {name = da_road_colony duration = -1 }
			set_province_flag = da_road_colony
		}
		
		if = {
			limit = {
				has_province_flag = da_thaig_ruin
			}
			clr_province_flag = da_thaig_ruin
			add_province_modifier = {name = da_thaig_colony duration = -1 }
			set_province_flag = da_thaig_colony
		}
		
		if = {
			limit = {
				has_province_flag = da_great_thaig_ruin
			}
			clr_province_flag = da_great_thaig_ruin
			add_province_modifier = {name = da_great_thaig_colony duration = -1 }
			set_province_flag = da_great_thaig_colony
		}
	}
	
	option = {
		
	}
}
